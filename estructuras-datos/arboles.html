Árboles

¿Qué son los árboles?

Los árboles son estructuras de datos jerárquicas que consisten en nodos conectados por aristas. Cada nodo puede tener hijos, y el nodo superior se llama raíz.

Operaciones Básicas

  • insert(valor) → agregar un nodo
  • delete(valor) → eliminar un nodo
  • search(valor) → buscar un nodo
  • height() → altura del árbol
  • isBalanced() → verificar si el árbol está balanceado

Tipos de recorridos

  1. Inorden (Izquierda, Raíz, Derecha)

    Se utiliza en árboles binarios de búsqueda para obtener los elementos en orden ascendente.

  2. Preorden (Raíz, Izquierda, Derecha)

    Útil para clonar árboles o para generar expresiones prefijas.

  3. Postorden (Izquierda, Derecha, Raíz)

    Usado en evaluaciones de árboles de expresión o para eliminar nodos.

Implementaciones Comunes

  • Expresiones matemáticas
  • Organización jerárquica
  • Índices en bases de datos
Animación: Cómo funcionan los árboles
Ejemplo en PHP

// Clase Nodo: cada nodo puede tener múltiples hijos
class Nodo {
    public $valor;  // Dato del nodo
    public $hijos;  // Lista de nodos hijos

    public function __construct($valor) {
        $this->valor = $valor;
        $this->hijos = [];
    }

    // Agrega un nodo hijo a este nodo
    public function agregarHijo($nodo) {
        $this->hijos[] = $nodo;
    }
}

// Construir el árbol:   A
//                      / \
//                     B   C
//                     |
//                     D
$raiz  = new Nodo("A");
$nodoB = new Nodo("B");
$nodoC = new Nodo("C");
$nodoD = new Nodo("D");

$raiz->agregarHijo($nodoB);   // A -> B
$raiz->agregarHijo($nodoC);   // A -> C
$nodoB->agregarHijo($nodoD);  // B -> D

// Recorrido preorden (raíz primero): A B D C
function recorrerArbol($nodo) {
    echo $nodo->valor . " ";
    foreach ($nodo->hijos as $hijo) {
        recorrerArbol($hijo); // Llamada recursiva
    }
}

recorrerArbol($raiz); // Imprime: A B D C
Ejemplo en JavaScript

// Clase Nodo: cada nodo puede tener múltiples hijos
class Nodo {
    constructor(valor) {
        this.valor = valor;  // Dato del nodo
        this.hijos = [];     // Lista de nodos hijos
    }

    // Agrega un nodo hijo a este nodo
    agregarHijo(nodo) {
        this.hijos.push(nodo);
    }
}

// Construir el árbol:   A
//                      / \
//                     B   C
//                     |
//                     D
const raiz  = new Nodo("A");
const nodoB = new Nodo("B");
const nodoC = new Nodo("C");
const nodoD = new Nodo("D");

raiz.agregarHijo(nodoB);   // A -> B
raiz.agregarHijo(nodoC);   // A -> C
nodoB.agregarHijo(nodoD);  // B -> D

// Recorrido preorden (raíz primero): A B D C
function recorrerArbol(nodo) {
    console.log(nodo.valor);
    nodo.hijos.forEach(hijo => recorrerArbol(hijo)); // Recursión
}

recorrerArbol(raiz); // Imprime: A, B, D, C
Ejemplo en Python

# Clase Nodo: cada nodo puede tener múltiples hijos
class Nodo:
    def __init__(self, valor):
        self.valor = valor  # Dato del nodo
        self.hijos = []     # Lista de nodos hijos

    def agregar_hijo(self, nodo):
        self.hijos.append(nodo)  # Añade un nodo hijo

# Construir el árbol:   A
#                      / \
#                     B   C
#                     |
#                     D
raiz   = Nodo("A")
nodo_b = Nodo("B")
nodo_c = Nodo("C")
nodo_d = Nodo("D")

raiz.agregar_hijo(nodo_b)   # A -> B
raiz.agregar_hijo(nodo_c)   # A -> C
nodo_b.agregar_hijo(nodo_d) # B -> D

# Recorrido preorden (raíz primero): A B D C
def recorrer_arbol(nodo):
    print(nodo.valor)
    for hijo in nodo.hijos:
        recorrer_arbol(hijo)  # Llamada recursiva

recorrer_arbol(raiz)  # Imprime: A, B, D, C
